#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL mod = 998244353;
inline void rd(LL &x)
{
	x=0;int f=1;char ch=getchar();
	while(ch<'0'||ch>'9'){if(ch=='-') f=-1;ch=getchar();}
	while(ch>='0'&&ch<='9'){x=x*10+ch-'0';ch=getchar();}
	x=x*f;
}
LL n,m,x[2],y[2],g,ans;
LL qpow(LL a)
{
	LL res=1,k=mod-2; a%=mod;
    while(k)
    {
        if(k&1)res=res*a%mod;
//        assert(res>=0); 
        a=a*a%mod;
        k>>=1;
    }
    return res;
}
LL _2 = qpow(2);
void down(LL x,LL h){
	x%=mod;h%=mod;	
//	assert(x*x%mod>=0); assert(x*x%mod*m%mod>=0);assert((x*x%mod*m%mod)*qpow(4*h));assert(qpow(4*h)>=0);
	ans += (x*x%mod*m%mod)*qpow(4*h);	
	ans%=mod;
}
void up(LL x,LL h)	{
	x%=mod;h%=mod;	
//	assert(m*h%mod>=0); assert(x*x%mod*m%mod>=0); assert((x*x%mod*m%mod)*qpow(4*h));assert(qpow(4*h)>=0); assert((m*h%mod) + ( (x*x%mod*m%mod) *qpow(4*h) )%mod>=0);
	ans += (m*h%mod) + ( (x*x%mod*m%mod) *qpow(4*h) )%mod;	
	ans%=mod;
}
void pi_4(LL x){
	x%=mod;	
//	assert(m*x%mod>=0);
	ans += (m*x%mod)*_2;	
	ans%=mod;
}
void solve()
{
	scanf("%lld%lld%lld",&n,&m,&g);
	rd(x[1]);	rd(y[1]);
	for(int i=2;i<=n;i++)
	{
		rd(x[i&1]);	rd(y[i&1]);
		if(y[i&1]>y[!(i&1)])	up(x[i&1]-x[!(i&1)],y[i&1]-y[!(i&1)]);
		if(y[i&1]<y[!(i&1)])	down(x[i&1]-x[!(i&1)],y[!(i&1)]-y[i&1]);
		if(y[i&1]==y[!(i&1)])	pi_4(x[i&1]-x[!(i&1)]);
	}
	printf("%lldJ",ans*g%mod);
}
int main()
{
	// freopen("jump.in","r",stdin);
	// freopen("jump.out","w",stdout);
	solve();
	fclose(stdin);
	fclose(stdout);
	return 0;
}

T2 放爆竹

题面（暂无）

by the way，题面描述是有误的。如果有两串雷完全相同那这整个题都 GG 了。

思路

首先，可以看到答案一定是不大于 2m2m2m 的。证明的话……咕了明天补。。。

因此我们可以将所有串循环至其长度为 1000，然后插入 Trie 中。答案即为 Trie 中最深的有 2 个 子节点的节点的深度。

Code

#include<bits/stdc++.h>
namespace imzzy{

const int maxn=20004,maxm=1001;
const int mod=239999999,inf=239999999;

char s[maxn][maxm];
int len[maxn];
class TRIE{ // 最常规的 Trie
	public: int ch[2][maxn*maxm],cnt,res;
	inline void insert(const char *str) {
		int now=0;
		for(rgi i=0;i<maxm;++i) {
			if(!ch[str[i]-'0'][now])
				ch[str[i]-'0'][now]=++cnt,res=std::max(res,(ch['1'-str[i]][now])?i:0);
			now=ch[str[i]-'0'][now];
		}
	} inline int query() {return res;}
	TRIE() {res=0,ch[0][0]=ch[1][0]=0;}
}trie;
signed main()
{P_INIT();
	// freopen("bomb.in","r",stdin);
	// freopen("bomb.out","w",stdout);
	int n; fin>>n;
	for(rgi i=1;i<=n;++i) {
		scanf("%s",s[i]),len[i]=strlen(s[i]);
		for(rgi j=0;j<maxm;++j) s[i][j]=s[i][j%len[i]]; // 重复到长度为 1000
		trie.insert(s[i]);
	} fout<<trie.query();
	return 0;
}

T3 pyy整队

题面（暂无）

by the way，题面描述还是有误的。“询问学生”本意指的“向 pyy 询问的学生”，而非“pyy 给出的答案对应的学生”，如此一说那岂不是输出 -1 就代表问 pyy 的不是个学生喽？（队首学生被剥夺学籍.jpg）

思路

用线段树维护区间最小值、平衡树（map）维护每个人站的位置（即对应下标）；从 111 到询问位置 −1-1−1 的位置之间年级排名最小的即为询问的答案。

第二问求至少移动多少位同学，使得队伍整齐，其实也就是所有连续 nnn 个位置中空缺最少的地方，然后枚举即可。

Code

const int maxn=200004;
const int mod=1000000007,inf=1201201201;

int a[maxn];
class segment_tree{
#define mid ((l+r)>>1)
private:
	std::map<int,int> pos;
	int size,qp,qv,*A,val[maxn<<2],indx;
	inline void pushup(int p) {val[p]=std::min(val[p<<1],val[p<<1|1]);}
	void build(int p,int l,int r) {
		if(l==r) return val[p]=A[l],void();
		build(p<<1,l,mid),build(p<<1|1,mid+1,r),pushup(p);
	}
	inline void modify(int p,int l,int r) { // 单点修改 qp 位置的值为 qv
		if(l==r) return val[p]=qv,void();
		if(qp<=mid) modify(p<<1,l,mid); else modify(p<<1|1,mid+1,r);
		pushup(p);
	}
	inline int query(int p,int l,int r) { // 区间查询 1~qp 的最小值
		if(r<=qp) return val[p];
		if(qp<=mid) return query(p<<1,l,mid);
		else return std::min(query(p<<1,l,mid),query(p<<1|1,mid+1,r));
	}
public:
	inline void build(int siz,int *k) {
		indx=size=siz,A=k,build(1,1,maxn-2);
		for(rgi i=1;i<=size;++i) pos[A[i]]=i;
	}
	inline void update(int v) {
		qp=pos[v],qv=inf,modify(1,1,maxn-2);
		qp=++indx,qv=v,modify(1,1,maxn-2);
		pos[v]=qp;
	}
	inline int query(int v) {
		if(!pos.count(v)||pos[v]==1) return -1;
		qp=pos[v]-1; int res=query(1,1,maxn-2);
		return res<inf?res:-1;
	}
	inline int query(int *tmp) { // tmp 只是个辅助数组（托）
		for(rgi i=1;i<=indx;++i) tmp[i]=0;
		for(std::map<int,int>::iterator it=pos.begin();it!=pos.end();++it)
			tmp[(*it).second]=1;
		int now=0,res=inf;
		for(rgi i=1;i<size;++i) now+=tmp[i];
		for(rgi i=size;i<=indx;++i) now+=tmp[i]-tmp[i-size],res=std::min(res,size-now);
		return res;
	}
#undef mid
}sgt;
signed main()
{P_INIT();
	// freopen("queue.in","r",stdin);
	// freopen("queue.out","w",stdout);
	rgi n,m,t; fin>>n>>m;
	for(rgi i=1;i<=n;++i) fin>>a[i];
	sgt.build(n,a);
	while(m--) {
		char c=' '; while(!isalpha(c)) c=getchar(); fin>>t;
		if(c=='A') fout<<sgt.query(t)<<endl;
		else sgt.update(t);
	} fout<<sgt.query(a);
	return 0;
}
