笔记 - 关于 RMQ

Author: imzzy1201 Date: 2020-06-06 Reading Time:2.8 mins words:473 Category: 笔记 Views: -

Make magic out of simple things.

下面给出一种 O(n)O(n)O(n) 时间预处理，O(log⁡4n)O(\log_4\sqrt n)O(log4​n​) 时间区间查询和单点修改（稍加更改，也可以写出区间修改），O(n)O(n)O(n) ，常数巨大的 RMQ 算法及代码实现

算法

分块套 ZKW 线段树

Code

const int maxn = 5000004, maxm = 2240, inf = 1000000007;
class RMQ
{
private:
	int siz, k[maxn], b;
	class ZKW
	{
	private:
		int siz, m, val[maxm << 2];
		inline void maintain(int x) {val[x] = std::min(val[x << 1], val[x << 1 | 1]);}
	public:
		inline void build(int x, int *A)
		{
			siz = x;
			for(m = 1; m < x; m <<= 1);
			for(int i = m + 1; i <= m + x; ++i) val[i] = A[i - m];
			for(int i = m - 1; i; --i) maintain(i);
		}
		inline void build(int x)
		{
			siz = x;
			for(m = 1; m < x; m <<= 1);
			for(int i = 1; i <= m + x; ++i) val[i] = 0;
		}
		inline void modify(int x, int v) {val[x += m] = v; for(x >>= 1; x; x >>= 1) maintain(x);}
		inline int query(int l, int r)
		{
			int res = inf;
			for(l = l + m - 1, r = r + m + 1; l ^ r ^ 1; l >>= 1, r >>= 1)
			{
				if(~l & 1) res = std::min(res, val[l ^ 1]);
				if(r & 1) res = std::min(res, val[r ^ 1]);
			}
			return res;
		}
		inline int query(int l) {return query(l, siz);}
		inline int query() {return query(1, siz);}
	} s[maxm];
public:
	inline void build(int size, int *val)
	{
		siz = size, b = sqrt(siz);
		int pre = 1;
		for(int i = 1; i <= siz + 1; ++i)
		{
			k[i] = (i - 1) / b + 1;
			if(k[i] != k[i - 1] || i == siz + 1)
				s[k[pre]].build(i - pre, val + pre - 1), pre = i;
		}
		s[0].build(k[siz]);
		for(int i = 1; i <= k[siz]; ++i) s[0].modify(i, s[i].query());
	}
	inline void modify(int x, int v) {s[k[x]].modify(x - b * (k[x] - 1), v), s[0].modify(k[x], s[k[x]].query());}
	inline int query(int l, int r)
	{
		if(r < l) std::swap(l, r);
		if(k[l] == k[r]) return s[k[l]].query(l - b * (k[l] - 1), r - b * (k[r] - 1));
		int res = std::min(s[k[l]].query(l - b * (k[l] - 1)), s[k[r]].query(1, r - b * (k[r] - 1)));
		res = std::min(res, s[0].query(k[l] + 1, k[r] - 1));
		return res;
	}
} R;
