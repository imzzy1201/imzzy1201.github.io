T2 迷路的刺豚

题面（暂无）

思路

可以看到 ppp 很小，所以可以状压当前已经经过了的菜店。

总共有以下步骤：

  1. 先动态规划求出刺豚在每个点的最大体积
  2. 用类似 SPFA 的 bfs 方法以每个菜店为起点求它到所有点的最短路以及最短路上的最大体积和（注意这里的体积和不包含起点那个菜店的体积）
  3. 状压当前已经到过的菜店，动态规划求出经过所有菜店的最短路以及最短路上的最大体积和

大概就是这样的，实现上有一些小细节吧。

	_f&&(_x=-_x,_f=0);return*this;}fastin(){_ch=_f=0;}

using namespace std;
const int maxn=304;
const int mod=1000000007,inf=1201201201;
const int dir[4][2]={{1,0},{0,1},{-1,0},{0,-1}};

int n,m,s,p;
int a[maxn][maxn],px[maxn],py[maxn]; // a 存地图，px py 存菜店的坐标（特别的，px[0] py[0] 存起点坐标）
int f[maxn][maxn],g[maxn][maxn]; // f[x][y] 是右下角坐标 (x,y) 的最大空正方形，g[x][y] 是中心为 (x,y) 的最大空正方形（即最大体积）
int to1[20][20],to2[20][20]; // 记录菜店两两之间的最短路、最短路对应的最大体积和
int dist[maxn][maxn],kkk[maxn][maxn]; // bfs 时分别记录最短路、最短路对应的最大体积和
queue<pair<int,int> >q; bool book[maxn][maxn];
void bfs(int t) {
	q.push(make_pair(px[t],py[t]));
	memset(dist,0x3f,sizeof(dist));
	memset(kkk,0x3f,sizeof(kkk));
	dist[px[t]][py[t]]=kkk[px[t]][py[t]]=0;
	while(!q.empty()) {
		pair<int,int> pos=q.front(); q.pop();
		int x=pos.first,y=pos.second;
		book[x][y]=0;
		for(rgi i=0;i<4;++i) {
			int tx=x+dir[i][0],ty=y+dir[i][1];
			if(tx>=0&&tx<=n&&ty>=0&&ty<=m&&!a[tx][ty])
			if(dist[tx][ty]>dist[x][y]+1 || dist[tx][ty]==dist[x][y]+1 && kkk[tx][ty]<kkk[x][y]+g[tx][ty]) {
				dist[tx][ty]=dist[x][y]+1,kkk[tx][ty]=kkk[x][y]+g[tx][ty];
				if(!book[tx][ty]) book[tx][ty]=1,q.push(make_pair(tx,ty));
			}
		}
	}
	for(rgi i=0;i<=p;++i) if(i!=t)
		to1[t][i]=dist[px[i]][py[i]],to2[t][i]=kkk[px[i]][py[i]];
}
int dp[34000][20],ccf[34000][20]; // dp[i][j] 是状态为 i、以 j 为终点的最短路，ccf 类似地记录最短路对应的最大体积和
signed main()
{P_INIT();
	// freopen("expand.in","r",stdin);
	// freopen("expand.out","w",stdout);
	fin>>n>>m>>s;
	for(rgi i=1;i<=n;++i) for(rgi j=1;j<=m;++j) fin>>a[i][j];
	for(rgi i=1;i<=n;++i) for(rgi j=1;j<=m;++j) if(a[i][j]==0)
		f[i][j]=min(min(f[i-1][j],f[i][j-1]),f[i-1][j-1])+1;
	for(rgi i=1;i<=n;++i) for(rgi j=1;j<=m;++j) for(rgi k=1;k<=s&&i+k<=n&&j+k<=m;++k)
		if(f[i+k][j+k]>=2*k+1) g[i][j]=k; else break;
	fin>>px[0]>>py[0]>>p,++px[0],++py[0];
	for(rgi i=1;i<=p;++i) fin>>px[i]>>py[i],++px[i],++py[i];
	for(rgi i=0;i<=p;++i) bfs(i);
	memset(dp,0x3f,sizeof(dp)),memset(dp[1],0,sizeof(dp[1]));
	memset(ccf,0,sizeof(ccf)),ccf[1][0]=g[px[0]][py[0]];
	for(rgi i=1;i<1<<(p+1);++i)
		for(rgi j=0;j<=p;++j) if(i&(1<<j))
			for(rgi k=0;k<=p;++k) if((i&(1<<k))&&j!=k)
				if(dp[i][j]>dp[i-(1<<j)][k]+to1[k][j] || dp[i][j]==dp[i-(1<<j)][k]+to1[k][j] && ccf[i][j]<ccf[i-(1<<j)][k]+to2[k][j])
					dp[i][j]=dp[i-(1<<j)][k]+to1[k][j],ccf[i][j]=ccf[i-(1<<j)][k]+to2[k][j];
	int ans1=inf,ans2=0;
	for(rgi i=0;i<=p;++i) {
		if(ans1>dp[(1<<(p+1))-1][i] || ans1==dp[(1<<(p+1))-1][i] && ans2<ccf[(1<<(p+1))-1][i])
			ans1=dp[(1<<(p+1))-1][i],ans2=ccf[(1<<(p+1))-1][i];
	} fout<<ans1<<' '<<ans2<<endl;
	return 0;
}
