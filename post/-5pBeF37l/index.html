class segment_tree{
#define mid ((l+r)>>1)
private:
	int size,ql,qr;
	int sum[2097154],sqrsum[2097154],tag[2097154];
    // 线段树维护 3 个值，sum 即区间和，sqrsum 即区间平方和，tag 即区间加的标记
	inline void pushdown(int p,int l,int r)
	{
		if(tag[p])
		{
			tag[p<<1]+=tag[p],tag[p<<1|1]+=tag[p];
			sqrsum[p<<1]=(1LL*sqrsum[p<<1] + 1LL*(mid-l+1)*tag[p]%mod*tag[p]%mod + 2LL*tag[p]*sum[p<<1]%mod)%mod;
			sqrsum[p<<1|1]=(1LL*sqrsum[p<<1|1] + 1LL*(r-mid)*tag[p]%mod*tag[p]%mod + 2LL*tag[p]*sum[p<<1|1]%mod)%mod;
            // 考虑 (a+b)²=a²+b²+2ab，所以区间加 b 后，每个数的平方值+=b²+2ab
            // 即区间平方和 += 区间长度×b²+2×区间和×b
			sum[p<<1]=(1LL*sum[p<<1] + 1LL*tag[p]*(mid-l+1))%mod;
			sum[p<<1|1]=(1LL*sum[p<<1|1] + 1LL*tag[p]*(r-mid))%mod;
			tag[p]=0;
		}
	}
	void update(int p,int l,int r)
	{
		if(l>=ql&&r<=qr)
		{
			sqrsum[p]=(1LL*sqrsum[p]+sum[p]*2LL+(r-l+1))%mod;
            // 同理，(a+1)²=a²+2a+1
            // 即区间平方和+=2×区间和+区间长度
			if((sum[p]+=r-l+1)>=mod) sum[p]-=mod;
			++tag[p]; return;
		}
		pushdown(p,l,r);
		if(mid>=ql) update(p<<1,l,mid);
		if(mid<qr) update(p<<1|1,mid+1,r);
		if((sum[p]=sum[p<<1]+sum[p<<1|1])>=mod) sum[p]-=mod;
		if((sqrsum[p]=sqrsum[p<<1]+sqrsum[p<<1|1])>=mod) sqrsum[p]-=mod;
	}
public:
	inline void build(int siz) {size=siz;}
	inline void update(int l,int r) {ql=l,qr=r,update(1,1,size);}
	inline int query() {return sqrsum[1];}
#undef mid
}sgt;

{

	rgi n,cnt; fin>>n;
	for(rgi i=1;i<=n;++i) fin>>a[i];
	memcpy(val,a,sizeof(int)*(n+1));
	std::sort(val+1,val+1+n),cnt=std::unique(val+1,val+1+n)-val;
	for(rgi i=1;i<=n;++i)
	{
		a[i]=std::lower_bound(val+1,val+cnt,a[i])-val;
		pre[i]=last[a[i]],last[a[i]]=i;
        // last[i] 当前最大的 j 满足 A[j]=i
	}
	sgt.build(n);
	rgi ans=0;
	for(rgi i=1;i<=n;++i)
	{
		sgt.update(pre[i]+1,i);
		if((ans+=sgt.query())>=mod) ans-=mod;
	}
	fout<<ans;
	return 0;
}

// cnt[i][j] 表示 i 的子树中，隶属于 j 的节点的个数；siz[i] 代表 i 的子树节点数
ll tmp[maxn],g[maxn][maxn],h[maxn];
void dfs(int fa,int now)
{
	cnt[now][s[now]=='1']=siz[now]=1,g[now][0]=1;
	for(rgi i=first[now],to=e[i].v;i;i=e[i].nxt,to=e[i].v)
	{
		if(to==fa) continue;
		dfs(now,to);
		for(rgi j=0;j<=siz[now]+siz[to];++j) tmp[j]=0;
		for(rgi j=0;g[now][j];++j) for(rgi k=0;g[to][k];++k)
			(tmp[j+k]+=g[now][j]*g[to][k])%=mod;
		for(rgi j=0;j<=siz[now]+siz[to];++j) g[now][j]=tmp[j];
		siz[now]+=siz[to],cnt[now][0]+=cnt[to][0],cnt[now][1]+=cnt[to][1];
	}
	// 上面这段相当于计算 g'
	
	for(rgi i=m;i>0;--i)
		(g[now][i]+=g[now][i-1]*std::max(0,cnt[now][s[now]=='0']-i+1))%=mod;
	// 上面这段则相当于计算真实的 g 值
}

ll ans[maxn];
ll fact[maxn],invfact[maxn];
inline ll inv(ll a)
{
	ll b=mod-2,res=1;
	while(b) {if(b&1) (res*=a)%=mod; (a*=a)%=mod,b>>=1;}
	return res;
}
inline ll C(ll a,ll b) {return fact[a]*invfact[b]%mod*invfact[a-b]%mod;}

inline void solve()
{
	fact[0]=invfact[0]=1;
	for(rgi i=1;i<=m;++i) fact[i]=fact[i-1]*i%mod,invfact[i]=inv(fact[i]);
	for(rgi i=0;i<=m;++i) h[i]=g[1][i]*fact[m-i]%mod/*,fout<<h[i]<<' '*/;
	for(rgi i=0;i<=m;++i) for(rgi j=i;j<=m;++j)
		(ans[i]+=h[j]*(((j-i)&1)?(mod-1):1)%mod*C(j,i))%=mod;
}
signed main()
{
	// freopen("match.in","r",stdin);
	// freopen("match.out","w",stdout);
	int x,y; fin>>n; scanf("%s",s+1);
	for(rgi i=1;i<n;++i) fin>>x>>y,addedge(x,y),addedge(y,x);
	m=n>>1,dfs(0,1),solve();
//	for(rgi i=1;i<=n;++i)
//	{
//		for(rgi j=0;j<=m;++j) fout<<g[i][j]<<' ';
//		fout<<endl;
//	}
	for(rgi i=0;i<=m;++i) fout<<ans[i]<<endl;
	return 0;
}
