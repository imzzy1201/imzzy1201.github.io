思路

考虑到其相当于求所有国王到目标位置的切比雪夫距离，而我们知道切比雪夫距离是可以转化为曼哈顿距离的。

举个例子下面坐标系中有 A,BA,BA,B 两点：

那么他们的切比雪夫距离就是 max⁡(∣xA−xB∣,∣yA−yB∣)=max⁡(∣1−5∣,∣4−2∣)=4\max(|x_A-x_B|,|y_A-y_B|)=\max(|1-5|,|4-2|)=4max(∣xA​−xB​∣,∣yA​−yB​∣)=max(∣1−5∣,∣4−2∣)=4（即红橙两段中较长的橙线段的长度）
它们的曼哈顿距离则为 ∣xA−xB∣+∣yA−yB∣=∣1−5∣+∣4−2∣=6|x_A-x_B|+|y_A-y_B|=|1-5|+|4-2|=6∣xA​−xB​∣+∣yA​−yB​∣=∣1−5∣+∣4−2∣=6（即红橙两线段长度和

这两种距离看起来没啥关系，其实却有很大的联系。

考虑旋转坐标轴，逆时针饶原点旋转 45°，也就是转到了这里：

在新的坐标轴上我们对两点求曼哈顿距离（即紫色折线的总的长度）

我们发现此时的曼哈顿距离 424\sqrt{2}42​ 正好是原来的切比雪夫距离 444 的 2\sqrt{2}2​ 倍。
多试几个点可以发现，这个结论是恒成立的。这个结论也不难证明，带入坐标推式子即可说明正确性。下面主要说的是实现方法。

实现的时候，可以直接将所有点（包括询问的点）的坐标 (x,y)(x,y)(x,y) 改成 (x+y,x−y)(x+y,x-y)(x+y,x−y)，然后求曼哈顿距离和即可。注意最终答案是这个距离和的一半。

关于求法的话，可以离散化后用树状数组对横纵坐标各维护两个值：区间内的点数和区间坐标和。统计答案时分别求大于和小于当前询问坐标的答案即可。代码应该很好懂：

#include<bits/stdc++.h>
#define rgi register int
#define ll long long
class fastin{
	private: int _ch,_f;
	public: inline fastin& operator>>(char&c) {c=getchar(); return *this;}
		template<typename _Tp> inline fastin& operator>>(_Tp&_x){
		_x=0; while(!isdigit(_ch)) _f|=(_ch==45),_ch=getchar();
		while(isdigit(_ch)) _x=(_x<<1)+(_x<<3)+(_ch^48),_ch=getchar();
		_f&&(_x=-_x,_f=0); return *this;}
	fastin() {_ch=_f=0;}
}fin;
class fastout{
#define endl '\n'
	private: int _num[32],_head;
	public: inline fastout& operator<<(char c) {putchar(c); return *this;}
		template<typename _Tp> inline fastout& operator<<(_Tp _x){
		_Tp _k; if(_x==0) {putchar('0'); return *this;} if(_x<0) putchar('-'),_x=-_x;
		while(_x>0) _k=_x/10,++_head,_num[_head]=(_x-(_k<<1)-(_k<<3))^48,_x=_k;
		while(_head>0) putchar(_num[_head]),--_head; return *this;}
	fastout() {_head=0;}
}fout;

// using namespace std;
const int maxn=1000004;
const int mod=998244353,inf=1000000007;

struct POINT{int x,y;}p[maxn];
int val0[maxn],cnt0,val1[maxn],cnt1; // 离散化用的数组
class BIT{ // 树状数组
private: ll size,val[maxn];
public: inline int lowbit(int x) {return x&-x;}
	inline void update(int p,ll x) {while(p<=size) val[p]+=x,p+=lowbit(p);}
	inline ll query(int p) {ll res=0; while(p>0) res+=val[p],p-=lowbit(p); return res;}
	inline ll query(int l,int r) {return query(r)-query(l-1);}
	inline void build(int s) {size=s; memset(val,0,sizeof(ll)*(size+2));}
}B[2],C[2]; // B 记录坐标和，C 点的个数
// 上面那些变量 0 指横坐标，1 指纵坐标
signed main() {
	// freopen("chess.in","r",stdin);
	// freopen("chess.out","w",stdout);
	int n,x,y,qx,qy; ll ans=0;
	rgi T,q; fin>>T;
	while(T--) {
        cnt0=cnt1=0,fin>>n>>q;
		for(rgi i=1;i<=n;++i) fin>>p[i].x>>p[i].y,val0[++cnt0]=p[i].x+p[i].y,val1[++cnt1]=p[i].x-p[i].y;
		std::sort(val0+1,val0+1+cnt0),cnt0=std::unique(val0+1,val0+1+cnt0)-val0-1;
		std::sort(val1+1,val1+1+cnt1),cnt1=std::unique(val1+1,val1+1+cnt1)-val1-1;
		B[0].build(cnt0),C[0].build(cnt0);
		B[1].build(cnt1),C[1].build(cnt1);
        // 上为离散化
		for(rgi i=1,t3,t4;i<=n;++i) {
			t3=std::lower_bound(val0+1,val0+1+cnt0,p[i].x+p[i].y)-val0;
			t4=std::lower_bound(val1+1,val1+1+cnt1,p[i].x-p[i].y)-val1;
			B[0].update(t3,p[i].x+p[i].y),C[0].update(t3,1);
			B[1].update(t4,p[i].x-p[i].y),C[1].update(t4,1);
		} // 插入树状数组
		while(q--) {
            fin>>x>>y,ans=0;
			qx=std::lower_bound(val0+1,val0+1+cnt0,x+y)-val0;
			qy=std::lower_bound(val1+1,val1+1+cnt1,x-y)-val1;
			ans+=C[0].query(qx-1)*(x+y)-B[0].query(qx-1);
			ans+=B[0].query(qx,cnt0)-C[0].query(qx,cnt0)*(x+y); // 统计横坐标
			ans+=C[1].query(qy-1)*(x-y)-B[1].query(qy-1);
			ans+=B[1].query(qy,cnt1)-C[1].query(qy,cnt1)*(x-y); // 统计纵坐标
			fout<<ans/2<<endl;
		}
	}
	return 0;
}

using namespace std;
const int maxn=500004;
const int mod=998244353,inf=2000000007;

class segment_tree{
#define mid ((l+r)>>1)
private:
	int size,ql,qr,qv,*A,fill[maxn<<2],cnt; // fill 打的是取 max 标记
	pair<int,int> minv[maxn<<2],changed[maxn]; // first 维护最小值，second 维护最小值对应的下标
	inline void pushup(int p) {minv[p]=min(minv[p<<1],minv[p<<1|1]);}
	inline void pushdown(int p,int l,int r) {
		if(!fill[p]) return;
		fill[p<<1]=max(fill[p],fill[p<<1]),fill[p<<1|1]=max(fill[p],fill[p<<1|1]);
		minv[p<<1]=max(minv[p<<1],make_pair(fill[p],minv[p<<1].second));
		minv[p<<1|1]=max(minv[p<<1|1],make_pair(fill[p],minv[p<<1|1].second));
		fill[p]=0;
	}
	void build(int p,int l,int r) {
		fill[p]=0; if(l==r) return minv[p]=make_pair(A[l],l),void();
		build(p<<1,l,mid),build(p<<1|1,mid+1,r),pushup(p);
	}
	void modify(int p,int l,int r) { // 修改单点的值
		if(l==r) return fill[p]=qv,minv[p]=make_pair(qv,l),void();
		pushdown(p,l,r);
		if(ql<=mid) modify(p<<1,l,mid);
		else modify(p<<1|1,mid+1,r);
		pushup(p);
	}
	void update_(int p,int l,int r) { // 区间取 max
		if(l>=ql&&r<=qr) return minv[p]=max(minv[p],make_pair(qv,minv[p].second)),fill[p]=max(fill[p],qv),void();
		pushdown(p,l,r);
		if(ql<=mid) update_(p<<1,l,mid);
		if(qr>mid) update_(p<<1|1,mid+1,r);
		pushup(p);
	}
	pair<int,int> query(int p,int l,int r) { // 查询区间最小值
		if(l>=ql&&r<=qr) return minv[p]; pushdown(p,l,r);
		if(qr<=mid) return query(p<<1,l,mid);
		if(ql>mid) return query(p<<1|1,mid+1,r);
		return min(query(p<<1,l,mid),query(p<<1|1,mid+1,r));
	}
public:
	inline void build(int siz,int *k) {A=k,size=siz,cnt=0,build(1,1,size);}
	inline void update(int l,int r,int v) {ql=l,qr=r,qv=v,update_(1,1,size);}
	inline bool query(int l,int r,int x,int k,int *ans) {
		bool flag=1;
		for(rgi i=1;i<=k&&flag;++i) {
			ql=l,qr=r,changed[++cnt]=query(1,1,size); // 改成 inf 时要加入队列因为最后要还原的
			ql=changed[cnt].second,qv=inf,modify(1,1,size);
			if(changed[cnt].first<x) ans[i]=changed[cnt].first; else flag=0;
		} while(cnt) ql=changed[cnt].second,qv=changed[cnt--].first,modify(1,1,size);
		return flag;
	}
#undef mid
}sgt;
int a[maxn];
signed main() {
	// freopen("segtree.in","r",stdin);
	// freopen("segtree.out","w",stdout);
	int n,m,op,l,r,x,k; fin>>n;
	for(rgi i=1;i<=n;++i) fin>>a[i]; sgt.build(n,a); 
	for(fin>>m;m;--m) {
		fin>>op;
		if(op==1) fin>>l>>r>>x,sgt.update(l,r,x);
		else{
			fin>>l>>r>>x>>k,op=sgt.query(l,r,x,k,a);
			if(op) {for(rgi i=1;i<=k;++i) fout<<a[i]<<' '; fout<<endl;}
			else fout<<-1<<endl;
		}
	}
	return 0;
}

T3 简单树上路径

题面（暂无）

思路

神仙题

这题要把它转化为二分图匹配的模型。

首先倍增求出这几个点的 LCA（即他们集合的点），然后同样用 bitset 倍增求出每个人能拿到的特产。

考虑如何验证每个人买 xxx 个特产是否有方案。可以建立二分图，左边 c×xc\times xc×x 个节点，右边 mmm 个节点。左边的 c×xc\times xc×x 个节点代表共 ccc 个人，每个人有 xxx 个节点，即要选 xxx 种特产；右边 mmm 个节点代表 mmm 种特产。每个的拥有的节点都与它能拿到的特产连边，然后跑二分图匹配。如果最大匹配刚好是 c×xc\times xc×x 就说明每个人都能选出 xxx 个特产且互不相同，即有解，否则无解。

根据 Hall 定理，记这 ccc 个人构成的集合为 SSS，则有 xmax⁡=min⁡(f(S′)/∣S′∣) (S′∈S)x_{\max}=\min(f(S')/|S'|)\ (S'\in S)xmax​=min(f(S′)/∣S′∣) (S′∈S)，其中 f(S′)f(S')f(S′) 代表 S′S'S′ 中所有人能拿到的特产的总种数。

最后注意到题目中毒瘤的 256M256M256M 空间限制，直接开 20×30000020\times 30000020×300000 个 bitset 倍增会挂空间，所以要询问离线，然后一层一层计算倍增 2i2^i2i 次，并同时计算这一层倍增对每次询问的每个人的贡献。

#define rgi register int
#define ll long long
class fastin{
	private: int _ch,_f;
	public: inline fastin& operator>>(char&c) {c=getchar(); return *this;}

}fin;
class fastout{
#define endl '\n'
	private: int _num[32],_head;
	public: inline fastout& operator<<(char c) {putchar(c); return *this;}
		template<typename _Tp> inline fastout& operator<<(_Tp _x){
		_Tp _k; if(_x==0) {putchar('0'); return *this;} if(_x<0) putchar('-'),_x=-_x;
		while(_x>0) _k=_x/10,++_head,_num[_head]=(_x-(_k<<1)-(_k<<3))^48,_x=_k;

// using namespace std;
const int maxn=300004,maxq=50004,maxm=1004;
const int mod=998244353,inf=1000000007;

int anc[20][maxn],dep[maxn],c[maxq],v[6][maxq];
std::bitset<maxm> f[maxn],g[maxn],p[6][maxq],tmp; // f 记录第 i 次倍增的结果，g 则是第 i-1 次
inline int LCA(int p,int q) {
	if(dep[p]<dep[q]) std::swap(p,q);
	for(rgi i=19;i>=0;--i) if(dep[q]<=dep[p]-(1<<i)) p=anc[i][p];
	if(p==q) return p;
	for(rgi i=19;i>=0;--i) if(anc[i][p]!=anc[i][q]) p=anc[i][p],q=anc[i][q];
	return anc[0][p];
}
signed main() {
	// freopen("travel.in","r",stdin);
	// freopen("travel.out","w",stdout);
	int n,m,q,t,ans; fin>>n>>m>>q;
	anc[0][1]=dep[1]=1; for(rgi i=2;i<=n;++i) fin>>anc[0][i],dep[i]=dep[anc[0][i]]+1;
	for(rgi i=1;i<=n;++i) fin>>t,g[i][t]=1; for(rgi i=1;i<=n;++i) f[i]=g[anc[0][i]]; // 先初始化好第 0 层倍增的结果
	for(rgi j=1;j<20;++j) for(rgi i=1;i<=n;++i) anc[j][i]=anc[j-1][anc[j-1][i]];
	for(rgi i=1;i<=q;++i) {fin>>c[i];
		for(rgi j=1;j<=c[i];++j) fin>>v[j][i],p[j][i]=g[v[j][i]];
		v[0][i]=v[1][i]; for(rgi j=2;j<=c[i];++j) v[0][i]=LCA(v[j][i],v[0][i]);
	}
	for(rgi k=0;k<20;++k) {
		for(rgi i=1;i<=q;++i) for(rgi j=1;j<=c[i];++j)
			if((dep[v[j][i]]-dep[v[0][i]])&(1<<k)) // 二进制分解当前节点到目标节点的深度差，即可判断是否用到了这一层倍增
				p[j][i]|=f[v[j][i]],v[j][i]=anc[k][v[j][i]];
		memcpy(g,f,sizeof(f));
		for(rgi i=1;i<=n;++i) f[i]=g[i]|g[anc[k][i]];
	}
	for(rgi k=1;k<=q;++k) {
		ans=inf;
		for(rgi i=1;i<1<<c[k];++i) {tmp=0;
			for(rgi j=1;j<=c[k];++j) if(i&(1<<(j-1))) tmp|=p[j][k];
			ans=std::min(ans,(int)tmp.count()/__builtin_popcount(i));
		}
		fout<<(ans*c[k])<<endl;
	}
    return 0;
}
